import matplotlib as rc
import matplotlib.pyplot as plt

def plot_prec_rec(probabilities,precision,recall,figsize=(12,7)):
    
    rc.rcParams.update({'font.size': 16})
    fig,ax1 = plt.subplots(figsize=figsize)
    ax2 = plt.twinx(ax1)

    plt1 = ax1.plot(probabilities,precision,linewidth=3,color='#278CCB',label="Precision")
    ax1.plot(probabilities,precision,'.',linewidth=3,color='#278CCB',markersize=15)
    plt2 = ax2.plot(probabilities,recall,linewidth=3,color='#EB9532',label="Recall")
    ax2.plot(probabilities,recall,'.',linewidth=3,color='#EB9532',markersize=15)

    ax1.grid(True)

    ax1.set_title("Precision - Recall score")
    ax1.set_xlabel("Cutoff probability")
    ax1.set_ylabel("Precision")
    ax2.set_ylabel("Recall")
    
    plttot = plt1+plt2
    labels = [l.get_label() for l in plttot]
    ax1.legend(plttot,labels,loc='upper center',fancybox=True)
